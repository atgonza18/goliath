# GOLIATH Cron Jobs
# Timezone: US Central (CT) — ensure server TZ is set or use TZ= prefix
#
# Install: Update GOLIATH_ROOT below, then:
#   crontab cron-jobs/crontab.txt
# Or append:
#   crontab -l | cat - cron-jobs/crontab.txt | crontab -
#
# Prerequisites:
#   1. pip install -r telegram-bot/requirements.txt
#   2. Ensure REPORT_CHAT_ID is set in .env
#   3. Claude CLI installed and authenticated

GOLIATH_ROOT=/opt/goliath

# 6 PM CT — Daily scan of POD, Schedule, Constraints across all 12 projects
0 0 * * * cd $GOLIATH_ROOT && python3 cron-jobs/daily_scan.py >> cron-jobs/reports/cron.log 2>&1

# Midnight CT — Create daily date folder under DSC Constraints & Production Reports
0 6 * * * cd $GOLIATH_ROOT && python3 cron-jobs/daily_constraints_folder.py >> cron-jobs/reports/cron.log 2>&1

# 8 AM CT — Send latest report to analyst via Telegram
0 14 * * * cd $GOLIATH_ROOT && python3 cron-jobs/morning_report.py >> cron-jobs/reports/cron.log 2>&1
